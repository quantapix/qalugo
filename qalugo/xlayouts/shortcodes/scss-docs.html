(called in %q)" $n $f $.Page.Path -}} {{- end -}} {{- $x := replace (index $match 0) $start "" -}} {{- $y := replace $x $end "" -}} {{- if (ne $strip "false") -}} {{- $y = replace $y " !default" "" -}} {{- end -}} {{- highlight $y "scss" "" -}}
